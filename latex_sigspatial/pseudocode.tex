\begin{algorithmic}
\Function{\texttt{GET}}{i,j,z}
\If {$tile \lbrack i, j, z \rbrack \neq nil$}
    \State \Return $tile \lbrack i, j, z \rbrack$
 \Else
    \State $tile \lbrack i, j, z \rbrack \gets \Call{\texttt{render}}{i,j,z}$
    \State \Return $tile \lbrack i, j, z \rbrack$
\EndIf 
\EndFunction
\end{algorithmic}

\vspace{2ex}

\begin{algorithmic}
\Function{\texttt{PUT}}{basedata}
    \State $tiles \gets \Call{\texttt{affected-by}}{basedata}$
    \ForAll{$tile \in tiles$}
         \State \Call{\texttt{invalidate}}{$tile$}
    \EndFor
    \State \Call{\texttt{store}}{$basedata$}

\EndFunction
\end{algorithmic}
