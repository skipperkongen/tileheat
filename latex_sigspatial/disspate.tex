\newcounter{row}
\newcounter{col}

\newcommand\setrow[3]{
  \setcounter{col}{1}
  \foreach \n in {#1, #2, #3} {
    \edef\x{\value{col}}
    \edef\y{\value{row}}
    \node[anchor=center] at (\x, \y) {\n};
    \stepcounter{col}
    \stepcounter{col}
  }
  \stepcounter{row}
  \stepcounter{row}
}

\begin{document}
\begin{tikzpicture}[scale=1]

  \begin{scope}
    \draw[very thick,scale=2] (0, 0) grid (3, 3);

    \setcounter{row}{1}
    \setrow {} {} {}
    \setrow {}  {$ {(i, j, k)} $} {}
    \setrow {}  {} {}
  \end{scope}


\end{tikzpicture}
\end{document}
