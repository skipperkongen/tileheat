\newcommand\setrow[3]{
  \setcounter{col}{2}
  \foreach \n in {#1, #2, #3} {
    \edef\x{\value{col}}
    \edef\y{\value{row}}
    \node[anchor=center] at (\x, \y) {\n};
    \stepcounter{col}
    \stepcounter{col}
    \stepcounter{col}
    \stepcounter{col}    
  }
  \stepcounter{row}
  \stepcounter{row}
  \stepcounter{row}
  \stepcounter{row}  
}
